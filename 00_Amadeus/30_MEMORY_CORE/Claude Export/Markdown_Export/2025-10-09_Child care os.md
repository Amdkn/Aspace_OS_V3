### **Étape 3 : Agent Roster-Orchestrator (1h)**

**Fichier** `agents/RosterOrchestrator.js` :
```javascript
const { Pool } = require('pg');
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

class RosterOrchestrator {
  async checkAllRatios() {
    const query = `
      SELECT 
        r.id,
        r.name,
        r.age_group,
        r.legal_ratio,
        COUNT(DISTINCT s.id) FILTER (WHERE s.active = true) as staff_count,
        COUNT(DISTINCT c.id) FILTER (WHERE c.present = true) as children_count,
        CASE 
          WHEN COUNT(DISTINCT s.id) = 0 THEN 999
          ELSE ROUND(COUNT(DISTINCT c.id)::numeric / NULLIF(COUNT(DISTINCT s.id), 0), 2)
        END as actual_ratio
      FROM rooms r
      LEFT JOIN staff s ON r.id = s.room_id AND s.active = true
      LEFT JOIN children c ON r.id = c.room_id AND c.present = true
      GROUP BY r.id, r.name, r.age_group, r.legal_ratio
    `;
    
    const result = await pool.query(query);
    
    const ratios = result.rows.map(room => {
      const status = this.calculateStatus(room.actual_ratio, room.legal_ratio);
      return { ...room, status };
    });
    
    // Sauvegarder snapshot pour audit
    await this.saveSnapshot(ratios);
    
    return ratios;
  }
  
  calculateStatus(actual, legal) {
    if (actual > legal) return 'violation';
    if (actual > legal * 0.8) return 'warning';
    return 'compliant';
  }
  
  async saveSnapshot(ratios) {
    const values = ratios.map(r => 
      `(${r.id}, ${r.staff_count}, ${r.children_count}, ${r.actual_ratio}, ${r.legal_ratio}, '${r.status}')`
    ).join(',');
    
    await pool.query(`
      INSERT INTO ratio_snapshots (room_id, staff_count, children_count, actual_ratio, legal_ratio, status)
      VALUES ${values}
    `);
  }
  
  async toggleStaff(staffId, active) {
    await pool.query('UPDATE staff SET active = $1 WHERE id = $2', [active, staffId]);
    return this.checkAllRatios();
  }
  
  async toggleChild(childId, present) {
    await pool.query('UPDATE children SET present = $1 WHERE id = $2', [present, childId]);
    return this.checkAllRatios();
  }
}

module.exports = new RosterOrchestrator();
```
